def mse(a,b):
    return mean_squared_error(a,b)
def mae(a,b):
    return mean_absolute_error(a,b)
def r2(a,b):
    return r2_score(a,b)
def rd(a):
    return round(a,4)

def row(name,ytest, yhat):
    return [name, rd(r2(ytest, yhat)),rd(mae(ytest, yhat)),rd(mse(ytest, yhat))]
